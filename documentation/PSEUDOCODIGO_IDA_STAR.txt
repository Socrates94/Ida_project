
ALGORITMO IDA_STAR(inicio, objetivo)
    // 1. Inicialización de estadísticas y cálculo de cota inicial
    nodos_expandidos = 0
    limite_tiempo = max_time
    
    // El threshold inicial es puramente la heurística del nodo raíz
    threshold = heuristica(inicio, objetivo)
    
    // 2. Bucle principal de Profundización Iterativa (Iterative Deepening)
    MIENTRAS VERDADERO HACER
        
        // Estructuras para el camino actual y evitar ciclos en la misma rama DFS
        camino = [inicio]
        visitados = CONJUNTO_CONTIENE(inicio)
        
        // 3. Llamada a la Búsqueda en Profundidad (DFS) acotada
        resultado, nuevo_threshold = DFS_ACOTADO(camino, 0, threshold, visitados, objetivo)
        
        // 4. Analizar respuesta de DFS
        SI resultado NO ES NULO ENTONCES
            // Se encontró la meta
            movimientos = EXTRAER_MOVIMIENTOS(resultado)
            RETORNAR movimientos
        FIN SI
        
        SI nuevo_threshold == INFINITO ENTONCES
            // No hay más nodos por explorar o se agotó el tiempo
            RETORNAR NULO
        FIN SI
        
        // 5. Actualizar el umbral para la siguiente iteración
        threshold = nuevo_threshold
        
    FIN MIENTRAS
FIN ALGORITMO


ALGORITMO DFS_ACOTADO(camino, g, threshold, visitados, objetivo)
    nodo_actual = OBTENER_ULTIMO(camino)
    
    // f(n) = g(n) + h(n)
    f = g + heuristica(nodo_actual, objetivo)
    nodos_expandidos = nodos_expandidos + 1
    
    // Podar rama si excede el umbral actual
    SI f > threshold ENTONCES
        RETORNAR NULO, f
    FIN SI
    
    // Si es el nodo meta, retornar el camino y el threshold con el que se encontró
    SI nodo_actual == objetivo ENTONCES
        RETORNAR camino, threshold
    FIN SI
    
    // Para rastrear el mínimo costo de los nodos podados
    min_threshold = INFINITO
    
    // 1. Expandir nodo actual
    sucesores = GENERAR_HIJOS(nodo_actual)
    
    // 2. Ordenamiento opcional por heurística (favorece caminos más prometedores primero)
    ORDENAR sucesores POR heuristica(hijo, objetivo) ASCENDENTE
    
    // 3. Explorar hijos recursivamente
    PARA CADA hijo EN sucesores HACER
        
        // Evitar ciclos en el camino DFS actual
        SI hijo NO ESTÁ EN visitados ENTONCES
            visitados.AGREGAR(hijo)
            camino.AGREGAR(hijo)
            
            // Llamada recursiva incrementando el costo (g+1)
            resultado, t = DFS_ACOTADO(camino, g + 1, threshold, visitados, objetivo)
            
            SI resultado NO ES NULO ENTONCES
                // Meta encontrada en esta rama, propagar hacia arriba
                RETORNAR resultado, threshold
            FIN SI
            
            // Si no se encontró, actualizar el min_threshold para la próxima iteración profunda
            SI t < min_threshold ENTONCES
                min_threshold = t
            FIN SI
            
            // Backtracking: deshacer cambios para explorar el siguiente hermano
            camino.ELIMINAR_ULTIMO()
            visitados.REMOVER(hijo)
        FIN SI
        
    FIN PARA
    
    // Retornar fallo en esta rama y el umbral mínimo excedido
    RETORNAR NULO, min_threshold
FIN ALGORITMO


FUNCIÓN EXTRAER_MOVIMIENTOS(camino)
    movimientos = []
    
    PARA i DESDE 0 HASTA longitud(camino) - 2 HACER
        estado_actual = camino[i]
        estado_siguiente = camino[i+1]
        
        pos_vacia_actual = OBTENER_POSICION_VACIA(estado_actual)
        pos_vacia_sig = OBTENER_POSICION_VACIA(estado_siguiente)
        
        // Calcular diferencias de fila y columna
        delta_fila = pos_vacia_sig.fila - pos_vacia_actual.fila
        delta_columna = pos_vacia_sig.col - pos_vacia_actual.col
        
        SI delta_fila == -1 ENTONCES movimientos.AGREGAR('U')
        SINO SI delta_fila == 1 ENTONCES movimientos.AGREGAR('D')
        SINO SI delta_columna == -1 ENTONCES movimientos.AGREGAR('L')
        SINO SI delta_columna == 1 ENTONCES movimientos.AGREGAR('R')
        FIN SI
    FIN PARA
    
    RETORNAR movimientos
FIN FUNCIÓN
